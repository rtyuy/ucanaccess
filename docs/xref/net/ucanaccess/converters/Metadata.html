106         public boolean isValidValue(Object _val) {
107             if (_val == null) {
108                 return false;
109             } else if (type.isEnum()) {
110                 for (Object en : type.getEnumConstants()) {
111                     if (en == _val || en.toString().equalsIgnoreCase(_val.toString())) {
112                         return true;
113                     }
114                 }
115                 return false;
116             } else if (validValues.isEmpty()) {
117                 if (type == Boolean.class) {
118                     return "true".equalsIgnoreCase(_val.toString()) || "false".equalsIgnoreCase(_val.toString());
119                 } else if (type == Integer.class) {
120                     return Try.catching(() -> Integer.parseInt(_val.toString())).map(i -> true).orElse(false);
121                 }
122                 return true;
123             } else if (_val instanceof CharSequence) {
124                 return validValues.contains(_val.toString().toUpperCase());
125             } else {
126                 return validValues.contains(_val);
127             }
128         }
129 
130         public static Property parse(Object _val) {
131             if (_val == null) {
132                 return null;
133             }
134             String val = _val.toString().strip().toLowerCase();
135             for (Property prop : values()) {
136                 if (val.equalsIgnoreCase(prop.name())) {
137                     return prop;
138                 }
139             }
140             return null;
141         }

197     public void createMetadata() throws SQLException {
198         try (Statement st = conn.createStatement()) {
199             st.execute(SCHEMA);
200             st.execute(PROP);
201             st.execute(TABLES);

204             st.execute(COLUMNS_VIEW);
205             loadProp();
206         }

209     public void loadProp() throws SQLException {
210         try (PreparedStatement ps = conn.prepareStatement(
211             "INSERT INTO UCA_METADATA.PROP(NAME, MAX_LEN, DEFAULT_VALUE, DESCRIPTION) VALUES(?, ?, ?, ?)")) {
212             for (Property prop : Property.values()) {
213                 ps.setObject(1, prop.name());
214                 ps.setObject(2, prop.getMaxLen());
215                 ps.setObject(3, Optional.ofNullable(prop.getDefaultValue()).map(Object::toString).orElse(null));
216                 ps.setObject(4, prop.getDescription());
217                 ps.execute();
218             }
219         }
220     }
