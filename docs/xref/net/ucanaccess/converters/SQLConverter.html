110     private static void aliases(String sql, NormalizedSQL nsql) {
111         Matcher m = Patterns.SELECT_FROM_START.matcher(sql);
112         if (m.find()) {
113 
114             int init = m.end();
115             sql = sql.substring(init);
116             m = Patterns.SELECT_FROM_END.matcher(sql);
117             if (m.find()) {
118                 int end = m.start();
119                 sql = sql.substring(0, end);
120 
121                 for (m = Patterns.UNESCAPED_ALIAS.matcher(sql); m.find(); m = Patterns.UNESCAPED_ALIAS.matcher(sql)) {
122                     int e = m.end();
123 
124                     sql = sql.substring(e) + " ";
125                     char[] sqlc = sql.toCharArray();
126                     if (sqlc[0] == '[') {
127                         continue;
128                     }
129                     StringBuilder sb = new StringBuilder();
130                     for (char c : sqlc) {
131                         if (c == ' ' || c == '\n' || c == '\r' || c == ',') {
132                             String key = preEscapingIdentifier(sb.toString());
133                             nsql.put(key, sb.toString());
134                             break;
135                         } else {
136                             sb.append(c);
137                         }
138                     }
139                 }
140 
141             }
142         }
143     }

159     private static int[] getQuoteGroup(String _s) {
160         if (!_s.contains("''")) {
161             Matcher m = Patterns.QUOTE_M.matcher(_s);
162             if (m.find()) {
163                 return new int[] {m.start(), m.end()};
164             }
165         } else {
166             int[] ret = new int[] {-1, -1};
167             Matcher m = Patterns.QUOTE_S.matcher(_s);
168             while (m.find()) {
169                 int start = m.start();
170                 int end = m.end();
171                 if ((end - start) % 2 == 0) {
172                     if (ret[0] == -1) {
173                         return new int[] {m.start(), m.end()};
174                     }
175                 } else {
176                     if (ret[0] == -1) {
177                         ret[0] = m.start();
178                     } else {
179                         ret[1] = m.end();
180                         return ret;
181                     }
182                 }
183             }
184         }
185         return new int[0];
186     }

188     private static int[] getDoubleQuoteGroup(String _s) {
189         if (!_s.contains("\"\"")) {
190             Matcher m = Patterns.DOUBLE_QUOTE_M.matcher(_s);
191             if (m.find()) {
192                 return new int[] {m.start(), m.end()};
193             }
194         } else {
195             int[] ret = new int[] {-1, -1};
196             Matcher mc = Patterns.DOUBLE_QUOTE_S.matcher(_s);
197             while (mc.find()) {
198                 int start = mc.start();
199                 int end = mc.end();
200                 if ((end - start) % 2 == 0) {
201                     if (ret[0] == -1) {
202                         return new int[] {mc.start(), mc.end()};
203                     }
204                     continue;
205                 } else {
206                     if (ret[0] == -1) {
207                         ret[0] = mc.start();
208                     } else {
209                         ret[1] = mc.end();
210                         return ret;
211                     }
212                 }
213             }
214         }
215         return new int[0];
216     }

356     public static NormalizedSQL convertSQL(String _sql, UcanaccessConnection _conn, boolean _creatingQuery) {
357         NormalizedSQL nsql = new NormalizedSQL();
358         String sql = _sql + " ";
359         aliases(sql, nsql);
360         sql = replaceBacktick(sql);
361         sql = replaceAposNames(sql);
362         sql = convertUnion(sql);
363         sql = convertAccessDate(sql);
364         sql = convertQuotedAliases(sql, nsql);
365         sql = escape(sql);
366         sql = convertLike(sql);
367         sql = replaceWhiteSpacedTables(sql);
368         // sql = replaceExclamationPoints(sql);
369         if (!_creatingQuery) {
370             Pivot.checkAndRefreshPivot(sql, _conn);
371             sql = DFunction.convertDFunctions(sql, _conn);
372         }
373         sql = sql.trim();
374 
375         nsql.setSql(sql);
376 
377         return nsql;
378     }

407     private static String convertQuotedAliases(String sql, NormalizedSQL nsql) {
408         for (Matcher m = Patterns.KIND_OF_SUBQUERY.matcher(sql); m.find(); m = Patterns.KIND_OF_SUBQUERY.matcher(sql)) {
409             String g2 = m.group(2).trim();
410             if (g2.endsWith(";")) {
411                 g2 = g2.substring(0, g2.length() - 1);
412             }
413             sql = sql.substring(0, m.start()) + "(" + g2 + ")" + sql.substring(m.end());
414         }
415         Set<String> hs = new HashSet<>();
416         String sqle = sql;
417         String sqlN = "";
418         for (Matcher m = Patterns.QUOTED_ALIAS.matcher(sqle); m.find(); m = Patterns.QUOTED_ALIAS.matcher(sqle)) {
419             String g2 = m.group(2);
420             if (g2.indexOf('\'') >= 0 || g2.indexOf('"') >= 0) {
421                 hs.add(g2);
422             }
423             String value = g2.substring(1, g2.length() - 1);
424             nsql.put(preEscapingIdentifier(value), value);
425             sqlN += sqle.substring(0, m.start()) + m.group(1) + g2.replaceAll("['\"]", "") + m.group(3);
426             sqle = sqle.substring(m.end());
427         }
428         sql = sqlN + sqle;
429         for (String escaped : hs) {
430             sql = sql.replaceAll("\\[" + escaped.substring(1, escaped.length() - 1) + "\\]",
431                     escaped.replaceAll("['\"]", ""));
432         }
433         return sql;
434     }

440     static void addWhiteSpacedTableNames(String _name) {
441         if (_name == null) {
442             return;
443         }
444         String name = basicEscapingIdentifier(_name);
445         if (name == null || WHITE_SPACED_TABLE_NAMES.contains(name)) {
446             return;
447         }
448         for (String alrIn : WHITE_SPACED_TABLE_NAMES) {
449             if (name.contains(alrIn)) {
450                 WHITE_SPACED_TABLE_NAMES.add(WHITE_SPACED_TABLE_NAMES.indexOf(alrIn), name);
451                 return;
452             }
453         }
454         WHITE_SPACED_TABLE_NAMES.add(name);
455     }
