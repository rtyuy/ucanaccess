5   /**
6    * Enum of MS Access versions to encapsulate Jackcess {@link FileFormat}.
7    */
8   public enum AccessVersion {
9   
10      V2000(FileFormat.V2000),
11      V2003(FileFormat.V2003),
12      V2007(FileFormat.V2007),
13      V2010(FileFormat.V2010),
14      V2016(FileFormat.V2016);
15  
16      private static final AccessVersion DEFAULT_ACCESS_VERSION = V2003;
17  
18      private final FileFormat           fileFormat;
19  
20      AccessVersion(FileFormat _fileFormat) {
21          fileFormat = _fileFormat;
22      }
23  
24      public FileFormat getFileFormat() {
25          return fileFormat;
26      }
27  
28      @Override
29      public String toString() {
30          return name().toLowerCase();
31      }
32  
33      public static AccessVersion getDefaultAccessVersion() {
34          return DEFAULT_ACCESS_VERSION;
35      }
36  
37      public static AccessVersion parse(String _str) {
38          if (_str == null) {
39              return null;
40          }
41          String val = _str.strip();
42          for (AccessVersion ver : values()) {
43              if (val.equalsIgnoreCase(ver.name())) {
44                  return ver;
45              }
46          }
47          return null;
48      }
