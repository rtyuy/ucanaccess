9   public class CreateIndexCommand implements ICommand {
10      private final String indexName;
11      private final String tableName;
12      private final String execId;
13  
14      public CreateIndexCommand(String _indexName, String _tableName, String _execId) {
15          indexName = _indexName;
16          tableName = _tableName;
17          execId = _execId;
18      }
19  
20      @Override
21      public String getExecId() {
22          return execId;
23      }
24  
25      @Override
26      public String getTableName() {
27          return tableName;
28      }
29  
30      @Override
31      public CommandType getType() {
32          return CommandType.DDL;
33      }
34  
35      @Override
36      public IFeedbackAction persist() throws SQLException {
37          Try.catching(() -> new Persist2Jet().createIndex(tableName, indexName))
38              .orThrow(UcanaccessSQLException::new);
39          return null;
40      }
41  
42      @Override
43      public IFeedbackAction rollback() {
44          return null;
45      }
46  }
