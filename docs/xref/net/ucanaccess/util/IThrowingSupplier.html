1   package net.ucanaccess.util;
2   
3   import java.util.Objects;
4   import java.util.function.Function;
5   import java.util.function.Supplier;
6   
7   /**
8    * A supplier of a result of type {@code R} which might throw an exception of type {@code T}.
9    * <p>
10   * This is a <a href="package-summary.html">functional interface</a> whose functional method is {@link #get()}.
11   *
12   * @param <R> the type of the result supplied
13   * @param <T> the type of Throwable thrown
14   *
15   * @author Markus Spann
16   * @since v5.1.0
17   */
18  @FunctionalInterface
19  public interface IThrowingSupplier<R, T extends Throwable> {
20  
21      /**
22       * Gets a result.
23       *
24       * @return a result
25       */
26      R get() throws T;
27  
28      /**
29       * Creates a {@link IThrowingSupplier} from a {@link Supplier}.
30       * @param <R> the type of the result supplied
31       * @param <T> the type of Throwable thrown
32       * @param _supplier a conventional supplier
33       * @return a throwing supplier
34       */
35      static <R, T extends Throwable> IThrowingSupplier<R, T> of(Supplier<R> _supplier) {
36          return () -> _supplier.get();
37      }
38  
39      /**
40       * Returns the supplied value, unless {@link #get()} threw an exception,
41       * in which case the value produced by the supplying function is returned.
42       *
43       * @param _function non-null function executed in case of exception. The exception occurred in the supplier is passed to this function
44       * @return a value
45       */
46      default R or(Function<Throwable, R> _function) {
47          Objects.requireNonNull(_function, "Function required");
48          try {
49              return get();
50          } catch (Throwable _ex) {
51              return _function.apply(_ex);
52          }
53      }
54  
55      /**
56       * Returns a {@link Supplier} for this throwing supplier.
57       * @return supplier
58       */
59      @SuppressWarnings("PMD.UseDiamondOperator")
60      default Supplier<R> toSupplier() {
61          return new Supplier<R>() {
62              @Override
63              public R get() {
64                  try {
65                      return IThrowingSupplier.this.get();
66                  } catch (Throwable _ex) {
67                      throw doThrow(_ex);
68                  }
69              }
70  
71              @SuppressWarnings("unchecked")
72              <T1 extends Throwable> RuntimeException doThrow(Throwable _t) throws T1 {
73                  throw (T1) _t; // perform vacuous cast
74              }
75  
76          };
77      }
78  
79  }
