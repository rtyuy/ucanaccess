53      @Override
54      public Connection connect(String _url, Properties _props) throws SQLException {
55          if (!acceptsURL(_url)) {
56              return null;
57          }
58  
59          Map<String, String> unknownProps = new LinkedHashMap<>();
60          Map<Property, String> props = readProperties(_props, _url,
61              (k, v) -> {
62                  unknownProps.put(k, v);
63                  LOGGER.log(Level.WARNING, "Unknown driver property {0} with value {1}", k, v);
64              });
65  
66          int idxSemicolon = _url.indexOf(';');
67          String fileDbPath = idxSemicolon > 0 ? _url.substring(URL_PREFIX.length(), idxSemicolon) : _url.substring(URL_PREFIX.length());
68          File fileDb = new File(fileDbPath);
69          DBReferenceSingleton as = DBReferenceSingleton.getInstance();
70  
71          synchronized (UcanaccessDriver.class) {
72              try {
73                  Session session = new Session();
74  
75                  boolean alreadyLoaded = as.loaded(fileDb);
76                  FileFormat ff = null;
77                  if (props.containsKey(newDatabaseVersion) && !fileDb.exists()) {
78                      ff = FileFormat.parse(props.get(newDatabaseVersion));
79                  }
80                  boolean useCustomOpener = props.containsKey(jackcessOpener);
81  
82                  IJackcessOpenerInterface jko = useCustomOpener
83                      ? newJackcessOpenerInstance(props.get(jackcessOpener))
84                      : new DefaultJackcessOpener();
85                  DBReference dbRef = alreadyLoaded ? as.getReference(fileDb)
86                      : as.loadReference(fileDb, ff, jko, props.get(password));
87  
88                  if (!alreadyLoaded) {
89                      if ((useCustomOpener
90                          || props.containsKey(encrypt) && Boolean.parseBoolean(props.get(encrypt)))
91                          && (props.containsKey(memory) && !Boolean.parseBoolean(props.get(memory))
92                              || props.containsKey(keepMirror))) {
93                          dbRef.setEncryptHSQLDB(true);
94                      }
95  
96                      if (props.containsKey(memory)) {
97                          dbRef.setInMemory(Boolean.parseBoolean(props.get(memory)));
98                      }
99  
100                     if (props.containsKey(lobScale)) {
101                         Integer vl = validateLobScale(props.get(lobScale));
102                         dbRef.setLobScale(vl);
103                     }
104 
105                     if (props.containsKey(keepMirror)) {
106                         dbRef.setInMemory(false);
107                         if (dbRef.isEncryptHSQLDB()) {
108                             LOGGER.log(Level.WARNING, "{0} parameter cannot be combined with parameters {1} or {2}, {3} skipped",
109                                 keepMirror, jackcessOpener, encrypt, keepMirror);
110                         } else {
111                             File dbMirror =
112                                 new File(props.get(keepMirror) + fileDb.getName().toUpperCase().hashCode());
113                             dbRef.setToKeepHsql(dbMirror);
114                             if (props.containsKey(readOnlyMirror)) {
115                                 dbRef.setMirrorReadOnly(Boolean.parseBoolean(props.get(readOnlyMirror)));
116                             }
117                         }
118                     }
119 
120                     if (props.containsKey(showSchema)) {
121                         dbRef.setShowSchema(Boolean.parseBoolean(props.get(showSchema)));
122                     }
123                     if (props.containsKey(inactivityTimeout)) {
124                         int millis = 60000 * Integer.parseInt(props.get(inactivityTimeout));
125                         dbRef.setInactivityTimeout(millis);
126                     }
127                     if (props.containsKey(singleConnection)) {
128                         dbRef.setImmediatelyReleaseResources(Boolean.parseBoolean(props.get(singleConnection)));
129                     }
130                     if (props.containsKey(immediatelyReleaseResources)) {
131                         dbRef.setImmediatelyReleaseResources(
132                             Boolean.parseBoolean(props.get(immediatelyReleaseResources)));
133                     }
134                     if (props.containsKey(lockMdb)) {
135                         dbRef.setOpenExclusive(Boolean.parseBoolean(props.get(lockMdb)));
136                     }
137 
138                     if (props.containsKey(openExclusive)) {
139                         dbRef.setOpenExclusive(Boolean.parseBoolean(props.get(openExclusive)));
140                     }
141 
142                     if (props.containsKey(concatNulls)) {
143                         dbRef.setConcatNulls(Boolean.parseBoolean(props.get(concatNulls)));
144                     }
145                     if (props.containsKey(preventReloading)) {
146                         dbRef.setPreventReloading(Boolean.parseBoolean(props.get(preventReloading)));
147                     }
148                     if (props.containsKey(reMap)) {
149                         Map<String, String> map = Arrays.stream(props.get(reMap).split("&")).map(s -> s.split("\\|")).filter(arr -> arr.length == 2)
150                             .collect(Collectors.toMap(k1 -> k1[0], v1 -> v1[1], (v1, v2) -> v1, LinkedHashMap::new));
151                         dbRef.setExternalResourcesMapping(map);
152                     }
153                     if (props.containsKey(supportsAccessLike)) {
154                         SQLConverter.setSupportsAccessLike(Boolean.parseBoolean(props.get(supportsAccessLike)));
155                     }
156                     if (props.containsKey(columnOrder)
157                         && ColumnOrder.DISPLAY == ColumnOrder.parse(props.get(columnOrder))) {
158                         dbRef.setColumnOrderDisplay();
159                     }
160                     if (props.containsKey(mirrorFolder) && dbRef.getToKeepHsql() == null) {
161                         dbRef.setInMemory(false);
162                         String fd = props.get(mirrorFolder);
163                         if ("java.io.tmpdir".equals(fd)) {
164                             fd = System.getProperty("java.io.tmpdir");
165                         }
166                         dbRef.setMirrorFolder(new File(fd));
167                     }
168                     if (props.containsKey(ignoreCase)) {
169                         dbRef.setIgnoreCase(Boolean.parseBoolean(props.get(ignoreCase)));
170                     }
171 
172                     dbRef.getDbIO().setErrorHandler((cl, bt, location, ex) -> {
173                         if (cl.getType().isTextual()) {
174                             LOGGER.log(Level.WARNING, "Invalid textual value in table {0}, column {1}: it might look like {2}",
175                                 cl.getTable().getName(), cl.getName(), new String(bt));
176                         }
177                         throw new IOException(ex);
178                     });
179                 }
180                 String pwd = dbRef.getDbIO().getDatabasePassword();
181                 if (pwd != null && !props.containsKey(jackcessOpener)) {
182                     if (!pwd.equals(props.get(password))) {
183                         throw new AuthenticationException();
184                     }
185 
186                 } else if (props.containsKey(jackcessOpener)) {
187                     String mpwd = props.get(password);
188                     session.setPassword(mpwd);
189                 }
190 
191                 Optional.ofNullable(props.get(user))
192                     .ifPresent(session::setUser);
193 
194                 SQLWarning sqlw = null;
195                 if (!alreadyLoaded) {
196                     boolean toBeLoaded = !dbRef.loadedFromKeptMirror(session);
197 
198                     Connection conn = dbRef.getHSQLDBConnection(session);
199                     // from version 2.7 hsqldb translates timestamps stored without timezone in the database
200                     // into the default timezone. MS Access however does not know timezones, therefore assume timestamps are UTC
201                     Try.withResources(conn::createStatement, st -> {
202                         st.executeQuery("SET TIME ZONE 'UTC'");
203                     }).orThrow();
204 
205                     LoadJet la = new LoadJet(conn, dbRef.getDbIO());
206                     if (props.containsKey(sysSchema)) {
207                         boolean val = Boolean.parseBoolean(props.get(sysSchema));
208                         dbRef.setSysSchema(val);
209                         la.setSysSchema(val);
210 
211                     }
212                     if (props.containsKey(skipIndexes)) {
213                         boolean val = Boolean.parseBoolean(props.get(skipIndexes));
214                         dbRef.setSkipIndexes(val);
215                         la.setSkipIndexes(val);
216                     }
217 
218                     if (toBeLoaded) {
219                         la.loadDB();
220                     } else {
221                         la.resetFunctionsDefault();
222                     }
223                     as.put(fileDb.getAbsolutePath(), dbRef);
224                     sqlw = la.getLoadingWarnings();
225                 }
226 
227                 Properties newProps = new Properties();
228                 props.forEach((key, value) -> newProps.put(key.name(), value));
229                 newProps.putAll(unknownProps);
230 
231                 UcanaccessConnection uc = new UcanaccessConnection(as.getReference(fileDb), newProps, session);
232                 uc.addWarnings(sqlw);
233                 uc.setUrl(_url);
234                 return uc;
235             } catch (Exception _ex) {
236                 throw new UcanaccessSQLException(_ex);
237             }
238         }
239     }

241     private Integer validateLobScale(String _property) {
242         try {
243             int i = Integer.parseInt(_property);
244 
245             if (i == 1 || i == 2 || i == 4 || i == 8 || i == 16 || i == 32) {
246                 return i;
247             }
248         } catch (Exception _ignored) {
249         }
250         LOGGER.log(Level.WARNING, "Lobscale value must equal at least one of the following values: 1,2,4,8,16,32, skipping it");
251         return null;
252     }

279     private IJackcessOpenerInterface newJackcessOpenerInstance(String className) throws UcanaccessSQLException {
280         Object instance = Try.catching(() -> Class.forName(className).getConstructor().newInstance()).orThrow(ex -> new UcanaccessSQLException("Failed to instantiate " + className, ex));
281 
282         if (instance instanceof IJackcessOpenerInterface) {
283             return (IJackcessOpenerInterface) instance;
284         }
285         throw new UcanaccessSQLException("Jackess Opener class must implement " + IJackcessOpenerInterface.class.getName());
286     }

288     /**
289      * Normalizes properties received as input and found in the driver url and returns a map of typed properties.<br>
290      * The method differentiates between known and unknown properties by return known properties as typed properties
291      * with their value and calling an optional consumer for each property key and value.<br>
292      * If a property is found multiple times or found in both, the input properties as well as in the driver url,
293      * the url has precedence over the input properties and the latest occurrence overall will be used.
294      *
295      * @param _input input properties
296      * @param _url driver url
297      * @param _unknownConsumer consumer of unknown property key/value
298      * @return map of unknown and normalized properties
299      */
300     static Map<Property, String> readProperties(Properties _input, String _url, BiConsumer<String, String> _unknownConsumer) {
301         Objects.requireNonNull(_input, "Properties required");
302         Objects.requireNonNull(_url, "URL required");
303 
304         Map<Property, String> props = new EnumMap<>(Property.class);
305 
306         for (String key : _input.stringPropertyNames()) {
307             Property prop = parse(key);
308             String val = _input.getProperty(key);
309             if (prop == null) {
310                 _unknownConsumer.accept(key, val);
311             } else {
312                 props.put(prop, val);
313             }
314         }
315 
316         Arrays.stream(_url.split(";"))
317             .skip(1)
318             .map(s -> s.split("=")).forEach(arr -> {
319                 Property prop = parse(arr[0]);
320                 String val = arr.length > 1 ? arr[1].strip() : null;
321                 if (prop == null) {
322                     _unknownConsumer.accept(arr[0], val);
323                 } else {
324                     props.put(prop, val);
325                 }
326             });
327 
328         return props;
329     }
