45          switch (getCommandType()) {
46          case BATCH:

49              return ((PreparedStatement) w).executeUpdate();
50          case NO_ARGUMENTS:
51              return w.executeUpdate(getSql());

54          case WITH_AUTO_GENERATED_KEYS:
