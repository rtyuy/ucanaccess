48          private ParameterReset(String _methodName, Class<?>[] _argClasses, Object... _args) {
49              methodName = _methodName;
50              args = _args;
51              argClasses = _argClasses;
52          }
53  
54          void execute() {
55              Try.catching(() -> {
56                  Method mth = PreparedStatement.class.getDeclaredMethod(methodName, argClasses);
57                  mth.invoke(wrapped, args);
58                  if (args[1] instanceof StringReader) {
59                      StringReader sr = (StringReader) args[1];
60                      sr.reset();
61                  }
62                  if (args[1] instanceof InputStream

64                      ((InputStream) args[1]).reset();
65                  }
66              }).orThrow(UcanaccessRuntimeException::new);
67          }
